module and_gate(
    input a, b,
    output y
);
    and g1(y, a, b);
endmodule


module tb_and_gate;
    reg a;
    reg b;
    wire y;

    and_gate dut (.a(a), .b(b), .y(y));

    initial begin
        $monitor("time=%0t | a=%b b=%b | y=%b", $time, a, b, y);

        a = 0; b = 0; #10;
        a = 0; b = 1; #10;
        a = 1; b = 0; #10;
        a = 1; b = 1; #10;

        $finish;
    end
endmodule
